```javascript
    // ── 포맷 헬퍼 ──
    function formatKRW(n) {
        const abs = Math.abs(n);
        const sign = n < 0 ? '-' : '';
        if (abs >= 1e8) return sign + '₩' + (abs / 1e8).toFixed(2) + '억';
        if (abs >= 1e4) return sign + '₩' + Math.round(abs / 1e4).toLocaleString('ko-KR') + '만';
        return sign + '₩' + Math.round(abs).toLocaleString('ko-KR');
    }

    function formatPct(n, showSign = true) {
        const sign = showSign && n > 0 ? '+' : '';
        return sign + n.toFixed(1) + '%';
    }

    function formatKRWDiff(n) {
        const sign = n > 0 ? '+' : '';
        return sign + formatKRW(n);
    }
```

```javascript
    // ── Yahoo Finance 데이터 페치 ──
    // range: '1mo'|'3mo'|'6mo'|'1y'|'3y'|'5y'
    // startDate, endDate: 'YYYY-MM-DD' 또는 null (null이면 range 사용)
    // 반환: { timestamps: number[], prices: number[], sym }
    // 실패 시 throw Error(메시지)
    async function fetchStockData(sym, range, startDate, endDate) {
        let yahooUrl;
        if (startDate && endDate) {
            const p1 = Math.floor(new Date(startDate).getTime() / 1000);
            const p2 = Math.floor(new Date(endDate).getTime() / 1000) + 86400;
            yahooUrl = `https://query1.finance.yahoo.com/v8/finance/chart/${encodeURIComponent(sym)}?period1=${p1}&period2=${p2}&interval=1d`;
        } else {
            yahooUrl = `https://query1.finance.yahoo.com/v8/finance/chart/${encodeURIComponent(sym)}?range=${range}&interval=1d`;
        }

        const url = CORS_PROXY + encodeURIComponent(yahooUrl);
        const res = await fetch(url);
        if (!res.ok) throw new Error(`${sym}: 서버 오류 (${res.status})`);

        const json = await res.json();
        const result = json?.chart?.result?.[0];
        if (!result) throw new Error(`${sym}: 데이터를 찾을 수 없습니다`);

        const timestamps = result.timestamp; // Unix 초 단위
        const prices = result.indicators?.adjclose?.[0]?.adjclose
                    ?? result.indicators?.quote?.[0]?.close;
        if (!timestamps || !prices || timestamps.length === 0) {
            throw new Error(`${sym}: 해당 기간에 거래 데이터가 없습니다`);
        }

        // null 가격 필터링 (거래정지일 등)
        const filtered = timestamps.reduce((acc, ts, i) => {
            if (prices[i] != null) {
                acc.timestamps.push(ts * 1000); // ms 변환
                acc.prices.push(prices[i]);
            }
            return acc;
        }, { timestamps: [], prices: [] });

        if (filtered.prices.length < 2) {
            throw new Error(`${sym}: 유효한 데이터가 부족합니다`);
        }

        return { sym, timestamps: filtered.timestamps, prices: filtered.prices };
    }
```

- [ ] **Step 1: calcLumpSum 함수를 `<script>` 블록에 추가**

```javascript
    // ── 일괄투자 계산 ──
    // 반환: { returns: number[], values: number[], dates: string[],
    //         finalValue, cumulReturn, totalInvested, cagr }
    function calcLumpSum(timestamps, prices, amount) {
        const shares = amount / prices[0];
        const values = prices.map(p => shares * p);
        const returns = values.map(v => (v / amount - 1) * 100);
        const dates = timestamps.map(ts => new Date(ts).toISOString().slice(0, 10));

        const finalValue = values[values.length - 1];
        const cumulReturn = returns[returns.length - 1];
        const years = (timestamps[timestamps.length - 1] - timestamps[0]) / (365.25 * 24 * 3600 * 1000);
        const cagr = years > 0 ? ((finalValue / amount) ** (1 / years) - 1) * 100 : 0;

        return { returns, values, dates, finalValue, cumulReturn, totalInvested: amount, cagr };
    }
```

```javascript
    // ── 적립식(DCA) 계산 ──
    // monthlyAmount: 월 투자금
    // 반환: { returns: number[], values: number[], dates: string[],
    //         finalValue, cumulReturn, totalInvested, cagr }
    function calcDCA(timestamps, prices, monthlyAmount) {
        // 매월 첫 번째 거래일 인덱스 찾기
        const monthlyIdx = new Set();
        let lastMonth = -1;
        timestamps.forEach((ts, i) => {
            const m = new Date(ts).getMonth() + new Date(ts).getFullYear() * 12;
            if (m !== lastMonth) { monthlyIdx.add(i); lastMonth = m; }
        });

        let totalShares = 0;
        let totalInvested = 0;
        const values = [];
        const returns = [];

        prices.forEach((p, i) => {
            if (monthlyIdx.has(i)) {
                totalShares += monthlyAmount / p;
                totalInvested += monthlyAmount;
            }
            const v = totalShares * p;
            values.push(v);
            returns.push(totalInvested > 0 ? (v / totalInvested - 1) * 100 : 0);
        });

        const dates = timestamps.map(ts => new Date(ts).toISOString().slice(0, 10));
        const finalValue = values[values.length - 1];
        const cumulReturn = returns[returns.length - 1];
        const years = (timestamps[timestamps.length - 1] - timestamps[0]) / (365.25 * 24 * 3600 * 1000);
        const cagr = years > 0 && totalInvested > 0
            ? ((finalValue / totalInvested) ** (1 / years) - 1) * 100 : 0;

        return { returns, values, dates, finalValue, cumulReturn, totalInvested, cagr };
    }
```

```javascript
    function renderChart(dates, resA, resB, symA, symB, mode) {
        _chartData = { dates, resA, resB, symA, symB };
        if (btChartInstance) btChartInstance.destroy();

        const isReturns = mode === 'returns';
        const dataA = isReturns ? resA.returns : resA.values;

        const MAX_PTS = 500;
        function downsample(arr) {
            if (arr.length <= MAX_PTS) return arr;
            const step = arr.length / MAX_PTS;
            return Array.from({ length: MAX_PTS }, (_, i) => arr[Math.round(i * step)]);
        }
        const sampledDates = downsample(dates);
        const sampledA = downsample(dataA);
        const sampledB = downsample(dataB);

                        backgroundColor: 'transparent',
                        borderWidth: 2,
```

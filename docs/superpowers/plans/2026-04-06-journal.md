    <script>
        // ── Storage ──
        const STORAGE_KEY = 'fg-journal';

        function loadData() {
            try { return JSON.parse(localStorage.getItem(STORAGE_KEY) || '{}'); }
            catch { return {}; }
        }

        function saveData(data) {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
        }

        // ── State ──
        let data = loadData();
        let viewYear  = new Date().getFullYear();
        let viewMonth = new Date().getMonth(); // 0-indexed
        let selectedDate = null; // "YYYY-MM-DD" or null
        let entryType = 'buy';

        // ── Helpers ──
        function dateKey(y, m, d) {
            return y + '-' + String(m + 1).padStart(2, '0') + '-' + String(d).padStart(2, '0');
        }

        function todayKey() {
            const n = new Date();
            return dateKey(n.getFullYear(), n.getMonth(), n.getDate());
        }

        function fmtPrice(v) {
            return Number(v).toLocaleString('ko-KR') + '원';
        }

        function fmtDate(key) {
            const [y, m, d] = key.split('-');
            return y + '년 ' + Number(m) + '월 ' + Number(d) + '일';
        }

        // ── Stats ──
        function updateStats() {
            let total = 0, monthBuy = 0, monthSell = 0;
            const symbols = new Set();
            const monthPrefix = viewYear + '-' + String(viewMonth + 1).padStart(2, '0');

            for (const [key, entries] of Object.entries(data)) {
                total += entries.length;
                if (key.startsWith(monthPrefix)) {
                    for (const e of entries) {
                        if (e.type === 'buy')  { monthBuy++;  symbols.add(e.symbol); }
                        if (e.type === 'sell') { monthSell++; symbols.add(e.symbol); }
                    }
                }
            }

            document.getElementById('stat-total').textContent      = total;
            document.getElementById('stat-month-buy').textContent  = monthBuy;
            document.getElementById('stat-month-sell').textContent = monthSell;
            document.getElementById('stat-symbols').textContent    = symbols.size;
        }

        // ── Calendar ──
        function renderCalendar() {
            document.getElementById('cal-title').textContent =
                viewYear + '년 ' + (viewMonth + 1) + '월';

            const grid = document.getElementById('cal-grid');
            grid.innerHTML = '';

            // Day-of-week headers (Sun–Sat)
            ['일','월','화','수','목','금','토'].forEach(d => {
                const el = document.createElement('div');
                el.className = 'cal-dow';
                el.textContent = d;
                grid.appendChild(el);
            });

            const firstDay = new Date(viewYear, viewMonth, 1).getDay(); // 0=Sun
            const daysInMonth = new Date(viewYear, viewMonth + 1, 0).getDate();
            const today = todayKey();

            // Leading blank cells
            for (let i = 0; i < firstDay; i++) {
                const el = document.createElement('div');
                el.className = 'cal-cell other-month';
                grid.appendChild(el);
            }

            for (let d = 1; d <= daysInMonth; d++) {
                const key = dateKey(viewYear, viewMonth, d);
                const entries = data[key] || [];
                const hasBuy  = entries.some(e => e.type === 'buy');
                const hasSell = entries.some(e => e.type === 'sell');

                const cell = document.createElement('div');
                cell.className = 'cal-cell';
                if (key === today)         cell.classList.add('today');
                if (key === selectedDate)  cell.classList.add('selected');

                const numEl = document.createElement('div');
                numEl.className = 'cal-day-num';
                numEl.textContent = d;
                cell.appendChild(numEl);

                if (hasBuy || hasSell) {
                    const dots = document.createElement('div');
                    dots.className = 'cal-dots';
                    const dot = document.createElement('div');
                    dot.className = 'cal-dot ' + (hasBuy && hasSell ? 'dot-both' : hasBuy ? 'dot-buy' : 'dot-sell');
                    dots.appendChild(dot);
                    cell.appendChild(dots);
                }

                cell.addEventListener('click', () => selectDay(key));
                grid.appendChild(cell);
            }

            // Trailing cells to complete last row
            const total = firstDay + daysInMonth;
            const trailing = total % 7 === 0 ? 0 : 7 - (total % 7);
            for (let i = 0; i < trailing; i++) {
                const el = document.createElement('div');
                el.className = 'cal-cell other-month';
                grid.appendChild(el);
            }
        }

        // ── Day panel ──
        function selectDay(key) {
            selectedDate = key;
            renderCalendar();
            renderDayPanel();
        }

        function renderDayPanel() {
            const panel = document.getElementById('day-panel');
            panel.style.display = 'block';
            document.getElementById('day-panel-title').textContent = fmtDate(selectedDate) + ' 거래 내역';

        // ── Type toggle ──
        function setType(t) {
            entryType = t;
            document.getElementById('btn-buy').className  = 'type-btn' + (t === 'buy'  ? ' active-buy'  : '');
            document.getElementById('btn-sell').className = 'type-btn' + (t === 'sell' ? ' active-sell' : '');

        // ── Add entry ──
        function addEntry() {
            const symbol = document.getElementById('f-symbol').value.trim();
            const price  = Number(document.getElementById('f-price').value);
            const qty    = Number(document.getElementById('f-qty').value);
            const memo   = document.getElementById('f-memo').value.trim();

            if (!symbol || !price || !qty) {
                alert('종목명, 가격, 수량을 입력해주세요.');
                return;
            }

            const entry = {
                id: Date.now() + '-' + Math.random().toString(36).slice(2, 7),
                type: entryType,
                symbol,
                price,
                qty,
                memo,
            };

            if (!data[selectedDate]) data[selectedDate] = [];
            data[selectedDate].push(entry);
            saveData(data);

            renderCalendar();
            renderDayPanel();
            updateStats();
        }

        // ── Delete entry ──
        function deleteEntry(id) {
            if (!selectedDate || !data[selectedDate]) return;
            data[selectedDate] = data[selectedDate].filter(e => e.id !== id);
            if (data[selectedDate].length === 0) delete data[selectedDate];
            saveData(data);
            renderCalendar();
            renderDayPanel();
            updateStats();
        }

        // ── Month nav ──
        document.getElementById('btn-prev').addEventListener('click', () => {
            viewMonth--;
            if (viewMonth < 0) { viewMonth = 11; viewYear--; }
            selectedDate = null;
            document.getElementById('day-panel').style.display = 'none';
            renderCalendar();
            updateStats();
        });

        document.getElementById('btn-next').addEventListener('click', () => {
            viewMonth++;
            if (viewMonth > 11) { viewMonth = 0; viewYear++; }
            selectedDate = null;
            document.getElementById('day-panel').style.display = 'none';
            renderCalendar();
            updateStats();
        });

        // ── Init ──
        renderCalendar();
        updateStats();
    </script>
</body>
